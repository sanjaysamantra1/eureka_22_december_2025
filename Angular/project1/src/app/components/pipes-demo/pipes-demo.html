Let's Learn Angular Pipes

uppercase , lowercase & titlecase pipes:
userName without pipe: {{ userName }}
userName with uppercase: {{ userName | uppercase }}
userName with lowercase: {{ userName | lowercase }}
userName with titlecase: {{ userName | titlecase }}
Currency pipe:
Salary: {{ mySal }}
Salary: {{ mySal | currency }}
Salary: {{ mySal | currency : 'INR' }}
Salary: {{ mySal | currency : '€' }}
Salary: {{ mySal | currency : '¥' }}
Salary: {{ mySal | currency : 'USD' : 'code' }}
Salary: {{ mySal | currency : 'USD' : 'symbol' }}
Date pipe:
{{ dateObj | date }}
{{ dateObj | date :'short'}}
{{ dateObj | date :'medium'}}
{{ dateObj | date :'long'}}
{{ dateObj | date :'full'}}
==================
{{ dateObj | date : 'shortDate' }}
{{ dateObj | date : 'mediumDate' }}
{{ dateObj | date : 'longDate' }}
{{ dateObj | date : 'fullDate' }}
=================
{{ dateObj | date : 'shortTime' }}
{{ dateObj | date : 'mediumTime' }}
{{ dateObj | date : 'longTime' }}
{{ dateObj | date : 'fullTime' }}
==============
{{ dateObj | date : 'dd-MMM-yyyy' }}
{{ dateObj | date : 'HH:mm' }}
{{ dateObj | date: "fullTime":"UTC" }}
Number & percent Pipe:
{{ 12.111222333 | number: "3.2-5" }}
{{ 12.1 | number: "3.2-5" }}
{{ 0.5 | number:'3.2-5' }}
------
{{ 0.5 | percent }}
{{ 0.75 | percent }}
{{ 0.04 | percent }}

}
